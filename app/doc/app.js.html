'use strict';

angular.module('demoApp', [
  'ngRoute',
  'demoControllers',
  'demoDirectives',
  'ngCookies'
]).config(['$routeProvider',
  function ($routeProvider) {
    $routeProvider.when('/login',{
      templateUrl:'demo/login.html',
      controller:'LoginCtrl'
    }).when('/register',{
      templateUrl:"demo/register.html",
      controller:"RegisterCtrl"
    }).otherwise({
      redirectTo:'/login'
    });
  }
]);
